def parte(izq: Int, der: Int): Int =
  val cen = izq + (der - izq) / 2
  val pivot = xs(cen)
  intercambiar(cen, der)   // mover pivot al final
  var p = izq
  for i <- izq until der do
    if xs(i) <= pivot then
      intercambiar(i, p)   // empujar elemento menor a la izquierda
      p += 1
   
  intercambiar(p, der)     // mover pivot a su posición final
  p                        // devolver posición final del pivot
